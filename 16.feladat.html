16. feladat: Legnagyobb közös osztó megkeresése

Írjon programot C programozási nyelvben két szám legnagyobb közös osztójának megkeresésére rekurzió segítségével.
Legnagyobb közös osztó megoldása C nyelvben
                
C kód:

#include<stdio.h>

int findGCD(int num1,int num2);
int main()
{
  int num1,num2,gcd;
  printf("\n\n Rekurzió : Két szám LKO-jának megtalálása :\n");
  printf("------------------------------------------\n");  
  printf(" Első szám: ");
  scanf("%d",&num1);
  printf(" Második szám: ");
  scanf("%d",&num2);
  
  gcd = findGCD(num1,num2);
  printf("\n Az LKO-ja a %d és %d-nek: %d\n\n",num1,num2,gcd);
  return 0;
}

int findGCD(int a,int b)
{
      while(a!=b)
      {
          if(a>b)
              return findGCD(a-b,b);
          else
              return findGCD(a,b-a);
      }
      return a;
}

19. feladat: Fibonacci-sorozat

Írjon programot C programozási nyelvben a fibonacci-sorozat kiszámítására rekurzió segítségével.
Fibonacci-sorozat kiszámítása C nyelvben
              
C kód:

#include <stdio.h>

int fibbonacci(int n) {
    if(n == 0){
      return 0;
    } else if(n == 1) {
      return 1;
    } else {
      return (fibbonacci(n-1) + fibbonacci(n-2));
    }
}

int main() {
    int n = 8;// a kívánt szám megadása
    int i;

    printf("A %d szám fibonacci sorozata: " , n);
  
    for(i = 0;i<n;i++) {
      printf("%d ",fibbonacci(i));            
    }
}

#include <stdio.h>

int factorial(int n) {
  //alapeset
  if(n == 0) {
      return 1;
  } else {
      return n * factorial(n-1);
  }
}

int main() {
