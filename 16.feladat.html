int findGCD(int num1,int num2);
int main()
{
  int num1,num2,gcd;
  printf("\n\n Rekurzió : Két szám LKO-jának megtalálása :\n");
  printf("------------------------------------------\n");  
  printf(" Első szám: ");
  scanf("%d",&num1);
  printf(" Második szám: ");
  scanf("%d",&num2);
  
  gcd = findGCD(num1,num2);
  printf("\n Az LKO-ja a %d és %d-nek: %d\n\n",num1,num2,gcd);
  return 0;
}

int findGCD(int a,int b)
{
      while(a!=b)
      {
          if(a>b)
              return findGCD(a-b,b);
          else
              return findGCD(a,b-a);
      }
      return a;
}

#include <stdio.h>

//Rekurzív függvény a beszúrásos rendezéshez
void recursiveInsertionSort(int arr[], int n){
    if (n <= 1)
      return;
    recursiveInsertionSort( arr, n-1 );
    int nth = arr[n-1];
    int j = n-2;
    while (j >= 0 && arr[j] > nth){
      arr[j+1] = arr[j];
      j--;
    }
    arr[j+1] = nth;
}

int fibbonacci(int n) {
    if(n == 0){
      return 0;
    } else if(n == 1) {
      return 1;
    } else {
      return (fibbonacci(n-1) + fibbonacci(n-2));
    }
}

20. feladat: Rúdvágás probléma

Írjon programot C programozási nyelvben a rúdvágás probléma megoldására rekurzió segítségével.
Rúdvágás probléma C nyelvben

//C program a rúdvágás problémára

#include <stdio.h>
#include <stdlib.h>
  
int max(int a, int b)
{
  return (a > b) ? a : b;
}
  
//Globális tömb a memoizálás céljából
int t[9][9];
  
/*Egy rekurzív program, amely a memoizálás 
segítségével valósítja meg a rúdvágási 
problémát (Top-Down)*/
int un_kp(int price[], int length[],
                      int Max_len, int n)
{
  
    /*a maximális ár nulla lesz, ha vagy 
      a rúd hossza nulla, vagy az ár nulla*/
    if (n == 0 || Max_len == 0)
    {
        return 0;
    }
  
    /*Ha a rúd hossza kisebb, mint a maximális hossz, 
      a Max_len figyelembe veszi.Most a nyereségtől 
      függően a Max_len vagy elfogadja, vagy elveti.*/
    if (length[n - 1] <= Max_len)
    {
        t[n][Max_len]
            = max(price[n - 1]
                      + un_kp(price, length,
                              Max_len - length[n - 1], n),
                  un_kp(price, length, Max_len, n - 1));
    }
  
    /*Ha a rúd hossza nagyobb, mint a megengedett méret,
      Max_len, akkor nem vesszük figyelembe.*/
    else
    {
        t[n][Max_len]
            = un_kp(price, length,
                              Max_len, n - 1);
    }
  
    /*A Max_len a kapott maximális értéket adja vissza, 
      Max_len, amely a Max_len oszlopban található.*/
    return t[n][Max_len];
}
  
/*program a fenti funkciók tesztelésére*/
int main()
{
    int price[] = { 1, 5, 8, 9, 10, 17, 17, 20 };
    int n = sizeof(price) / sizeof(price[0]);
    int length[n];
    for (int i = 0; i < n; i++)
    {
        length[i] = i + 1;
    }
    int Max_len = n;
  
    //Függvény hívás
    printf("A maximális kapott érték %d \n",
            un_kp(price, length, n, Max_len));
}
