28. feladat: Palindrom

Írjon programot C programozási nyelvben annak ellenőrzésére, hogy egy adott karakterlánc palindrom-e vagy sem.
Palindrom C nyelvben
              
C kód:

#include <stdio.h>
#include <string.h>
  
void checkPalindrome(char [], int);
  
int main()
{
    char wordPal[25];
    printf("\n\n Rekurzió : Ellenőrizze, hogy egy adott karakterlánc palindrom-e vagy sem: \n");
  printf("----------------------------------------------------------\n");	
  
    printf(" Adjon meg egy szót, hogy ellenőrizze a palindromát: ");
    scanf("%s", wordPal);
    checkPalindrome(wordPal, 0);//a Palindrom ellenőrzésére szolgáló függvény hívása
  
    return 0;
}
  
void checkPalindrome(char wordPal[], int index)
{
    int len = strlen(wordPal) - (index + 1);
    if (wordPal[index] == wordPal[len])
    {
        if (index + 1 == len || index == len)
        {
            printf(" A beírt szó palindrom.\n\n");
            return;
        }
        checkPalindrome(wordPal, index + 1);//a függvény önmagát hívja rekurzívan
    }
    else
    {
        printf("A beírt szó nem palindrom.\n\n");
    }
}
