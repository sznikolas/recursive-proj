long convertBinary(int decNo)
{
    static long biNo,r,fctor = 1;

    if(decNo != 0)
    {
          r = decNo % 2;
          biNo = biNo + r * fctor;
          fctor = fctor * 10;
          convertBinary(decNo / 2);//a convertBinary függvény hívja öngmagát rekurzívan
    }
    return biNo;
}

#include <stdio.h>
int lcmCalculate(int a, int b);

    scanf("%d", &n2);
// Biztosítja, hogy a LKT első paraméterének kisebbnek kell lennie, mint a másodiknak.
    if(n1 >  n2)
        lcmOf = lcmCalculate(n2, n1);//az lcmCalculate függvény hívása az LKT számításhoz
    else
        lcmOf = lcmCalculate(n1, n2);//az lcmCalculate függvény hívása az LKT számításhoz
    printf(" Az LKT-ja a %d és %d számnak :  %d\n\n", n1, n2, lcmOf);
    return 0;
}
int lcmCalculate(int a, int b)//az n1 és n2 értéke átadódik a és b értéknek
{
    static int m = 0;
    //Max érték hozzáadásával növeli m értékét
    m += b;
//  Ha talált egy közös többszöröst, akkor adja vissza a m-et
    if((m % a == 0) && (m % b == 0))
    {
        return m;
    }
    else
    {
        lcmCalculate(a, b);//az lcmCalculate hívja önmagát
    }
}

{
    int pos = 0;

        return 0;

25. feladat: Bináris keresés rekurzióval

Írjon programot C programozási nyelvben bináris keresésre rekurzióval.
Bináris keresés rekurzióval C nyelvben
              
C kód:

#include <stdio.h>
int binarySearch(int*, int, int, int, int);

int main()
{
  int arr1[10], i, n, md, c, low, hg;

  printf("\n\n Rekurzió : Bináris keresés :\n");
  printf("-----------------------------------\n");
  printf(" A tömbben tárolandó elemek számának megadása: ");
  scanf("%d", &n);
  printf(" Adja meg a %d elemet a tömbbe növekvő sorrendben :\n", n);
  for (i = 0; i < n; i++) 
  {
      printf(" elem - %d : ", i);
      scanf("%d", &arr1[i]);
  }
  printf(" Adja meg a keresendő számot : ");
  scanf("%d", &md);
  low = 0, hg = n - 1;
  c = binarySearch(arr1, n, md, low, hg);
  if (c == 0)
      printf(" A keresett szám nem létezik a tömbben.\n\n");
  else
      printf(" A keresett szám létezik a tömbben.\n\n");
  return 0;
}

int binarySearch(int arr1[], int n, int md, int low, int hg)
{
  int mid, c = 0;
  if (low <= hg) 
  {
      mid = (low + hg) / 2;
      if (md == arr1[mid]) 
      {
          c = 1;
      }
      else if (md < arr1[mid]) 
      {
          return binarySearch(arr1, n, md, low, mid - 1);
      }
      else
          return binarySearch(arr1, n, md, mid + 1, hg);
  }
  else
      return c;
}
