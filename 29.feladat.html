    void invertStringRecursive(char *source, char *dest, int l, int h)
    {
        if (h < 0)
        {
            dest[strlen(source)] = '\0';
            return;
        }
        else
        {
            dest[l] = source[h];
            invertStringRecursive(source, dest, ++l, --h);
        }
    }
