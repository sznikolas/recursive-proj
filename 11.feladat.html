12. feladat: Első N szám összege

Írjon programot C programozási nyelvben az 1-től n-ig terjedő számok összegének kiszámítására rekurzió segítségével.
Első N szám összegének megoldása C programozási nyelvben
                
#include<stdio.h>

int sumOfRange(int);
  
int main() 
{
    int n1;
    int sum;
    printf("\n\n Rekurzió : számítsuk ki az 1-től n-ig terjedő számok összegét: \n");
    printf("-----------------------------------------------------------\n");    
  
    printf(" Az 1-től kezdődő tartomány utolsó számának bevitele: ");
    scanf("%d", &n1);
  
    sum = sumOfRange(n1); 
    printf("\n Az 1-től terjedő számok összege %d-ig : %d\n\n", n1, sum);
  
    return (0);
}
  
int sumOfRange(int n1) 
{
    int res;
    if (n1 == 1) 
    {
      return (1);
    } else 
    {
      res = n1 + sumOfRange(n1 - 1); //a sumOfRange függvény meghívja öngmagát
    }
    return (res);
}

    ArrayElement(arr1, st+1, l);//az ArrayElement függvény meghívja önmagát

14. feladat: Egy adott n méretű tömb r elemének összes lehetséges kombinációjának kiírása

Írjon programot C programozási nyelvben egy adott n méretű tömb r elemének összes lehetséges kombinációjának kiírása rekurzió segítségével.
N méretű tömb r elemének összes kombinációja
                
// Program az összes r méretű kombináció kiíratására egy n méretű tömbben
#include <stdio.h>
void combinationUtil(int arr[], int data[], int start, int end,
                     int index, int r);
 
/* A fő függvény, amely kiírja az összes r méretű kombinációt
    az n méretű arr[]-ban. Ez a függvény elsősorban 
    a combinationUtil() függvényt használja */
void printCombination(int arr[], int n, int r)
{
    // Egy ideiglenes tömb az összes kombináció egyenkénti tárolására
    int data[r];
 
    // Az összes kombináció kiírása az ideiglenes 'data[]' tömb segítségével
    combinationUtil(arr, data, 0, n-1, 0, r);
}
 
/* arr[]  ---> Bemeneti tömb
   data[] ---> Ideiglenes tömb az aktuális kombináció tárolására
   start & end ---> Kezdő és vég indexek az arr[] tömbben
   index  ---> Aktuális index a data[] tömbben
   r ---> A kiíratandó kombináció mérete  */
void combinationUtil(int arr[], int data[], int start, int end,
                     int index, int r)
{
    // Az aktuális kombináció készen áll a kiíratásra, írja ki
    if (index == r)
    {
        for (int j=0; j<r; j++)
            printf("%d ", data[j]);
        printf("\n");
        return;
    }
 
    /* Cserélje ki az indexet az összes lehetséges elemre. Az "end-i+1 >= r-index" 
    feltétel biztosítja, hogy az indexnél lévő egyik elem 
    beillesztése kombinációt eredményezzen a 
    fennmaradó pozíciókban lévő többi elemmel. */
    for (int i=start; i<=end && end-i+1 >= r-index; i++)
    {
        data[index] = arr[i];
        combinationUtil(arr, data, i+1, end, index+1, r);
    }
}
 
// tesztelés és bemenet megadása
int main()
{
    int arr[] = {1, 2, 3, 4, 5, 9, 8};
    int r = 4;
    int n = sizeof(arr)/sizeof(arr[0]);
    printCombination(arr, n, r);
}

15. feladat: Egy n szám számjegyei összegének megszámlálása

Írjon programot C programozási nyelvben egy szám számjegyei összegének rekurzióval történő kiszámítására.
Egy n szám számjegyei összegének megszámlálása C programozási nyelvben
                
#include <stdio.h>

int DigitSum(int num);
  
int main()
{
    int n1, sum;
    printf("\n\n Rekurzió : Egy szám számjegyeinek összege :\n");
    printf("-----------------------------------------------------\n");     
    printf(" Adjon meg bármilyen számot, hogy megtalálja a számjegyek összegét: ");
    scanf("%d", &n1);
      
    sum = DigitSum(n1);//hívja a függvényt a számításhoz
      
    printf(" Az összege a %d = %d\n\n", n1, sum);
      
    return 0;
}

int DigitSum(int n1)
{
    if(n1 == 0)
        return 0;
          
    return ((n1 % 10) + DigitSum(n1 / 10));//a DigitSum függvény öngmagát hívja
}
