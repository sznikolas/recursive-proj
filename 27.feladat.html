    void multiplyMatrix(int arm[MAX][MAX],int brm[MAX][MAX])
    {
    
        static int sum,i=0,j=0,k=0;
    
        if(i<rone)
        { //row of first matrix
        if(j<ctwo)
        {  //column of second matrix
             if(k<cone)
             {
                 sum=sum+arm[i][k]*brm[k][j];
                 k++;
                 multiplyMatrix(arm,brm);
             }
             crm[i][j]=sum;
                 sum=0;
                 k=0;
                 j++;
                 multiplyMatrix(arm,brm);
        }
        j=0;
        i++;
        multiplyMatrix(arm,brm);
        }
