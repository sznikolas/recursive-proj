31. feladat: Másolás és beillesztés rekurzióval

Írjon programot C programozási nyelvben, amely rekurzióval másol át egy karakterláncot egy másikba.
Másolás és beillesztés megoldása C nyelvben
                
#include <stdio.h>

void copyString(char [], char [], int);
  
int main()
{
    char stng1[20], stng2[20];
    printf("\n\n Rekurzió: Egy karakterlánc másolása egy másikba: \n");
    printf("---------------------------------------------\n"); 	
  
    printf(" Adja meg a másolandó karakterláncot: ");
    scanf("%s", stng1);
    copyString(stng1, stng2, 0);
    printf("\n A karakterlánc sikeresen másolva.\n\n");
    printf(" Az első karakterlánc: %s\n", stng1);
    printf(" A másolt karakterlánc: %s\n\n", stng2);
    return 0;
}
  
void copyString(char stng1[], char stng2[], int ctr)
{
    stng2[ctr] = stng1[ctr];
    if (stng1[ctr] == '\0')
        return;
    copyString(stng1, stng2, ctr + 1);
}

Számsorozat reverse
                
// C program egy szám számjegyeinek visszafordítására
#include <stdio.h>;
 
/* Rekurzív függvény a num számjegyeinek visszafordítására*/
int reversDigits(int num)
{
    static int rev_num = 0;
    static int base_pos = 1;
    if (num > 0) {
        reversDigits(num / 10);
        rev_num += (num % 10) * base_pos;
        base_pos *= 10;
    }
    return rev_num;
}
 
/*Vezérlőprogram a reversDigits teszteléséhez*/
int main()
{
    int num = 459851662;
    printf("A megadott szám reverse: %d", reversDigits(num));
 
    getchar();
    return 0;
}

34. feladat: Bináris fa bejárása spirális sorrendben

Írjon programot C programozási nyelvben, amely egy fa spirális sorrendben történő bejárását kiírja. Használjon rekurziót!
Bináris fa bejárása spirális formában
                
// C program rekurzív szintbejárással spirális formában
#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>
 
   /* A bináris fa csomópontjainak adatai: pointer a bal oldali gyermekre 
   és egy pointer a jobb oldali gyermekre */
struct node {
    int data;
    struct node* left;
    struct node* right;
};
 
/* Függvény prototípusok */
void printGivenLevel(struct node* root, int level, int ltr);
int height(struct node* node);
struct node* newNode(int data);
 
/* Függvény egy fa spirális bejárásának kiírására*/
void printSpiral(struct node* root)
{
    int h = height(root);
    int i;
 
    /*ltr -> Left to Right. Ha ez a változó be van állítva,
      akkor az adott szinten balról jobbra haladunk. */
    bool ltr = false;
    for (i = 1; i <= h; i++) {
        printGivenLevel(root, i, ltr);
 
    /*Fordított ltr, hogy a következő szintet ellenkező sorrendben járja be*/
        ltr = !ltr;
    }
}
 
/* A csomópontok kiíratása az adott szinten */
void printGivenLevel(struct node* root, int level, int ltr)
{
    if (root == NULL)
        return;
    if (level == 1)
        printf("%d ", root->data);
    else if (level > 1) {
        if (ltr) {
            printGivenLevel(root->left, level - 1, ltr);
            printGivenLevel(root->right, level - 1, ltr);
        }
        else {
            printGivenLevel(root->right, level - 1, ltr);
            printGivenLevel(root->left, level - 1, ltr);
        }
    }
}
 
/*  A fa "magasságának" kiszámítása - a csomópontok száma 
    a gyökércsomóponttól a legtávolabbi levélcsomópontig 
    vezető leghosszabb út mentén.*/
int height(struct node* node)
{
    if (node == NULL)
        return 0;
    else {
        /* kiszámítja az egyes részfák magasságát */
        int lheight = height(node->left);
        int rheight = height(node->right);
 
        /* használja a nagyobbat*/
        if (lheight > rheight)
            return (lheight + 1);
        else
            return (rheight + 1);
    }
}
 
/*  Segédfüggvény, amely új csomópontot rendel ki a megadott adatokkal 
    és NULL bal és jobb pointerekkel.  */
struct node* newNode(int data)
{
    struct node* node = (struct node*)
        malloc(sizeof(struct node));
    node->data = data;
    node->left = NULL;
    node->right = NULL;
 
    return (node);
}
 
/* Vezérprogram a teszteléshez*/
int main()
{
    struct node* root = newNode(1);
    root->left = newNode(2);
    root->right = newNode(3);
    root->left->left = newNode(7);
    root->left->right = newNode(6);
    root->right->left = newNode(5);
    root->right->right = newNode(4);
    printf("Spirális bejárása a megadott bináris fának: \n");
    printSpiral(root);
 
    return 0;
}

Josephus-probléma
                
#include <stdio.h>

int josephus(int n, int k)
{
    if (n == 1) //base case
        return 1;
    else
        return (josephus(n - 1, k) + k - 1) % n + 1; //rekurzió
}
  
// tesztelés
int main()
{
    int n = 14;
    int k = 2;
    printf("A biztonságos hely: %d", josephus(n, k));
    return 0;
}
